# 前端安全与防护

## 什么是前端安全？

前端安全是指保护Web应用前端代码和用户数据免受各种攻击和威胁。

**英文原意**：Frontend Security
**中文理解**：给前端代码和用户数据加上"防盗门"和"保险箱"

**生活类比**：就像给房子安装防盗门、监控摄像头，防止小偷入侵；给重要文件加上密码锁，防止他人偷看。

## 常见安全威胁

### 1. 跨站脚本攻击（XSS）

**英文**：Cross-Site Scripting (XSS)

**定义**：攻击者向网页注入恶意脚本，当用户浏览时执行

**类型**：
- **存储型XSS**：恶意脚本存储在服务器数据库中
- **反射型XSS**：恶意脚本通过URL参数传递
- **DOM型XSS**：通过修改DOM结构注入脚本

```javascript
// 危险的代码示例（不要这样写）
// 1. 直接插入HTML
function displayUserInput(input) {
  document.getElementById('output').innerHTML = input; // 危险！
}

// 2. 使用eval执行用户输入
function executeUserCode(code) {
  eval(code); // 极度危险！
}

// 3. 不安全的URL跳转
function redirectTo(url) {
  window.location = url; // 可能被利用
}

// 安全的做法
// 1. 使用textContent而不是innerHTML
function safeDisplayUserInput(input) {
  const element = document.getElementById('output');
  element.textContent = input; // 安全！
}

// 2. 使用安全的解析方法
function safeParseJSON(jsonString) {
  try {
    return JSON.parse(jsonString); // 安全！
  } catch (error) {
    console.error('Invalid JSON:', error);
    return null;
  }
}

// 3. URL白名单验证
function safeRedirectTo(url) {
  const allowedDomains = ['example.com', 'trusted-site.com'];
  const urlObj = new URL(url);
  
  if (allowedDomains.includes(urlObj.hostname)) {
    window.location = url; // 安全！
  } else {
    console.error('Untrusted domain:', urlObj.hostname);
  }
}
```

### 2. 跨站请求伪造（CSRF）

**英文**：Cross-Site Request Forgery (CSRF)

**定义**：攻击者诱导用户在已登录的网站上执行非本意的操作

```javascript
// 易受CSRF攻击的代码
// 1. 简单的POST请求
function deleteUser(id) {
  fetch('/api/users/' + id, {
    method: 'DELETE'
  });
}

// 2. 没有验证的请求
function transferMoney(amount, toAccount) {
  fetch('/api/transfer', {
    method: 'POST',
    body: JSON.stringify({ amount, toAccount })
  });
}

// 防护CSRF的做法
// 1. 使用CSRF Token
class CSRFProtection {
  constructor() {
    this.token = this.getToken();
  }
  
  getToken() {
    // 从meta标签或cookie获取CSRF token
    const meta = document.querySelector('meta[name="csrf-token"]');
    return meta ? meta.content : '';
  }
  
  // 所有修改数据的请求都带上token
  makeSecureRequest(url, options = {}) {
    const headers = {
      ...options.headers,
      'X-CSRF-Token': this.token
    };
    
    return fetch(url, {
      ...options,
      headers,
      credentials: 'same-origin' // 确保cookie被发送
    });
  }
  
  deleteUser(id) {
    return this.makeSecureRequest('/api/users/' + id, {
      method: 'DELETE'
    });
  }
  
  transferMoney(amount, toAccount) {
    return this.makeSecureRequest('/api/transfer', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ amount, toAccount })
    });
  }
}

// 2. 双重Cookie验证
function doubleSubmitCookie() {
  // 生成随机token
  const token = generateRandomToken();
  
  // 设置cookie
  document.cookie = `csrf-token=${token}; Path=/; Secure; SameSite=Strict`;
  
  // 在请求中同时发送cookie和header
  return fetch('/api/secure-endpoint', {
    method: 'POST',
    headers: {
      'X-CSRF-Token': token,
      'Content-Type': 'application/json'
    },
    credentials: 'include' // 包含cookie
  });
}

// 3. SameSite Cookie
function setSecureCookie(name, value) {
  // SameSite属性可以防止CSRF攻击
  document.cookie = `${name}=${value}; Path=/; Secure; SameSite=Strict`;
}
```

### 3. 点击劫持（Clickjacking）

**英文**：Clickjacking

**定义**：攻击者通过透明的iframe覆盖在合法页面上，诱导用户点击

```javascript
// 防护点击劫持
// 1. 设置X-Frame-Options响应头（后端设置）
// X-Frame-Options: DENY 或 X-Frame-Options: SAMEORIGIN

// 2. 前端检测
function detectClickjacking() {
  // 检查是否在iframe中
  if (window.top !== window.self) {
    // 可以选择跳出iframe或显示警告
    if (confirm('此页面正在iframe中运行，是否跳出？')) {
      window.top.location = window.self.location;
    }
  }
  
  // 检测透明度
  const hasTransparentOverlay = () => {
    const elements = document.elementsFromPoint(100, 100);
    return elements.some(el => {
      const style = window.getComputedStyle(el);
      return parseFloat(style.opacity) < 0.1;
    });
  };
  
  if (hasTransparentOverlay()) {
    console.warn('检测到透明覆盖层，可能存在点击劫持风险');
  }
}

// 3. 使用Frame Buster脚本
function frameBuster() {
  const style = document.createElement('style');
  style.innerHTML = `
    html { display: none; }
  `;
  
  if (self === top) {
    document.documentElement.style.display = 'block';
  } else {
    top.location = self.location;
  }
}
```

```javascript
// src/utils/security/validator.js - 输入验证工具

class InputValidator {
  // 白名单验证
  static whitelist(value, allowedChars) {
    const regex = new RegExp(`[^${allowedChars}]`, 'g');
    return value.replace(regex, '');
  }
  
  // 邮箱验证
  static isEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
  }
  
  // URL验证
  static isUrl(url) {
    try {
      const urlObj = new URL(url);
      // 只允许HTTP和HTTPS协议
      return ['http:', 'https:'].includes(urlObj.protocol);
    } catch {
      return false;
    }
  }
  
  // 文件上传验证
  static validateFile(file, options = {}) {
    const {
      maxSize = 5 * 1024 * 1024, // 5MB
      allowedTypes = [],
      allowedExtensions = []
    } = options;
    
    // 检查文件大小
    if (file.size > maxSize) {
      throw new Error(`文件大小不能超过${maxSize / 1024 / 1024}MB`);
    }
    
    // 检查文件类型
    if (allowedTypes.length && !allowedTypes.includes(file.type)) {
      throw new Error(`不支持的文件类型: ${file.type}`);
    }
    
    // 检查文件扩展名
    const extension = file.name.split('.').pop().toLowerCase();
    if (allowedExtensions.length && !allowedExtensions.includes(extension)) {
      throw new Error(`不支持的文件扩展名: ${extension}`);
    }
    
    return true;
  }
  
  // SQL注入检测
  static detectSQLInjection(input) {
    const sqlPatterns = [
      /(\b(union|select|insert|update|delete|drop|create|alter|exec|execute|script|declare|truncate)\b)/gi,
      /(--|\/\*|\*\/|;|'|"|`)/g,
      /(\b(or|and)\b.*=.*\b(or|and)\b)/gi
    ];
    
    return sqlPatterns.some(pattern => pattern.test(input));
  }
  
  // XSS检测
  static detectXSS(input) {
    const xssPatterns = [
      /<script[^>]*>.*?<\/script>/gi,
      /javascript:/gi,
      /on\w+\s*=/gi, // onclick, onload等
      /<iframe[^>]*>.*?<\/iframe>/gi,
      /<object[^>]*>.*?<\/object>/gi
    ];
    
    return xssPatterns.some(pattern => pattern.test(input));
  }
}

// 使用示例
export function validateUserInput(input, type) {
  // 基本清理
  let cleaned = input.trim();
  
  // 根据类型进行特定验证
  switch (type) {
    case 'email':
      if (!InputValidator.isEmail(cleaned)) {
        throw new Error('请输入有效的邮箱地址');
      }
      return cleaned;
      
    case 'username':
      // 只允许字母、数字和下划线
      cleaned = InputValidator.whitelist(cleaned, 'a-zA-Z0-9_');
      if (cleaned.length < 3 || cleaned.length > 20) {
        throw new Error('用户名长度必须在3-20个字符之间');
      }
      return cleaned;
      
    case 'url':
      if (!InputValidator.isUrl(cleaned)) {
        throw new Error('请输入有效的URL地址');
      }
      return cleaned;
      
    default:
      // 通用清理
      if (InputValidator.detectXSS(cleaned)) {
        throw new Error('输入包含潜在的安全风险');
      }
      
      if (InputValidator.detectSQLInjection(cleaned)) {
        throw new Error('输入包含非法字符');
      }
      
      return cleaned;
  }
}
```

```javascript
// src/utils/security/csp.js - 内容安全策略

// CSP配置生成器
class CSPGenerator {
  constructor() {
    this.directives = {
      'default-src': ["'self'"],
      'script-src': ["'self'"],
      'style-src': ["'self'", "'unsafe-inline'"], // Element UI需要内联样式
      'img-src': ["'self'", 'data:', 'https:'],
      'font-src': ["'self'", 'data:'],
      'connect-src': ["'self'"],
      'frame-src': ["'none'"],
      'object-src': ["'none'"],
      'base-uri': ["'self'"],
      'form-action': ["'self'"],
      'frame-ancestors': ["'none'"]
    };
  }
  
  // 添加允许的源
  addSource(directive, source) {
    if (!this.directives[directive]) {
      this.directives[directive] = [];
    }
    this.directives[directive].push(source);
  }
  
  // 生成CSP头部
  generate() {
    return Object.entries(this.directives)
      .map(([directive, sources]) => `${directive} ${sources.join(' ')}`)
      .join('; ');
  }
  
  // 设置CSP头部
  setHeaders() {
    const cspHeader = this.generate();
    
    // 通过meta标签设置（仅用于开发环境）
    if (process.env.NODE_ENV === 'development') {
      const meta = document.createElement('meta');
      meta.httpEquiv = 'Content-Security-Policy';
      meta.content = cspHeader;
      document.head.appendChild(meta);
    }
    
    // 生产环境应该通过服务器设置
    console.log('CSP Header:', cspHeader);
    return cspHeader;
  }
}

// 使用示例
export function setupCSP() {
  const csp = new CSPGenerator();
  
  // 添加API域名
  csp.addSource('connect-src', 'https://api.example.com');
  csp.addSource('connect-src', 'https://cdn.example.com');
  
  // 添加图片CDN
  csp.addSource('img-src', 'https://img.example.com');
  
  // 添加字体CDN
  csp.addSource('font-src', 'https://fonts.googleapis.com');
  csp.addSource('font-src', 'https://fonts.gstatic.com');
  
  // 设置CSP
  return csp.setHeaders();
}
```

```javascript
class SecureStorage {
  constructor() {
    this.prefix = 'secure_';
    this.encryptionKey = this.deriveKey();
  }
  
  // 从用户密码派生加密密钥
  deriveKey() {

    return 'derived-key-from-user-password';
  }
  
  // 简单的加密（实际应用中应使用成熟的加密库）
  encrypt(data) {
    // 这里使用简单的base64编码作为示例
    // 实际应用中应使用AES等强加密算法
    return btoa(JSON.stringify(data));
  }
  
  decrypt(encryptedData) {
    try {
      return JSON.parse(atob(encryptedData));
    } catch (error) {
      console.error('解密失败:', error);
      return null;
    }
  }
  
  // 安全设置数据
  setItem(key, value, options = {}) {
    const { 
      encrypted = true,
      expireTime = null, // 过期时间（毫秒）
      secure = true // 只在HTTPS下可用
    } = options;
    
    // 检查安全要求
    if (secure && window.location.protocol !== 'https:') {
      console.warn('安全存储要求HTTPS环境');
      return false;
    }
    
    let dataToStore = value;
    
    // 添加过期时间
    const storageData = {
      value: value,
      expireTime: expireTime ? Date.now() + expireTime : null,
      createdAt: Date.now()
    };
    
    // 加密敏感数据
    if (encrypted) {
      dataToStore = this.encrypt(storageData);
    }
    
    try {
      localStorage.setItem(this.prefix + key, dataToStore);
      return true;
    } catch (error) {
      console.error('存储失败:', error);
      return false;
    }
  }
  
  // 安全获取数据
  getItem(key, options = {}) {
    const { encrypted = true } = options;
    
    try {
      const storedData = localStorage.getItem(this.prefix + key);
      
      if (!storedData) {
        return null;
      }
      
      let data = storedData;
      
      // 解密数据
      if (encrypted) {
        data = this.decrypt(storedData);
      }
      
      // 检查过期时间
      if (data && data.expireTime && Date.now() > data.expireTime) {
        this.removeItem(key);
        return null;
      }
      
      return data ? data.value : null;
    } catch (error) {
      console.error('读取失败:', error);
      return null;
    }
  }
  
  // 删除数据
  removeItem(key) {
    localStorage.removeItem(this.prefix + key);
  }
  
  // 清除所有安全存储的数据
  clear() {
    const keys = [];
    for (let i = 0; i < localStorage.length; i++) {
      const key = localStorage.key(i);
      if (key && key.startsWith(this.prefix)) {
        keys.push(key);
      }
    }
    
    keys.forEach(key => localStorage.removeItem(key));
  }
}
```

```javascript
// src/utils/security/communication.js - 安全通信

class SecureCommunication {
  constructor() {
    this.apiWhitelist = [
      'https://api.liuma-platform.com',
      'https://cdn.liuma-platform.com'
    ];
  }
  
  // 验证API地址
  validateApiUrl(url) {
    try {
      const urlObj = new URL(url);
      const isWhitelisted = this.apiWhitelist.some(allowed => 
        urlObj.origin === allowed
      );
      
      if (!isWhitelisted) {
        throw new Error(`未授权的API地址: ${url}`);
      }
      
      // 确保使用HTTPS
      if (urlObj.protocol !== 'https:') {
        throw new Error('API通信必须使用HTTPS');
      }
      
      return true;
    } catch (error) {
      console.error('API地址验证失败:', error);
      return false;
    }
  }
  
  // 安全的HTTP请求
  async secureRequest(url, options = {}) {
    // 验证URL
    if (!this.validateApiUrl(url)) {
      throw new Error('不安全的API请求');
    }
    
    // 安全默认配置
    const secureOptions = {
      ...options,
      credentials: 'include', // 包含cookies
      headers: {
        ...options.headers,
        'Content-Type': 'application/json',
        'X-Requested-With': 'XMLHttpRequest', // 防止CSRF
        'X-CSRF-Token': this.getCSRFToken(),
        'X-API-Version': '1.0'
      }
    };
    
    try {
      const response = await fetch(url, secureOptions);
      
      // 验证响应
      if (!this.validateResponse(response)) {
        throw new Error('响应验证失败');
      }
      
      return response;
    } catch (error) {
      console.error('安全请求失败:', error);
      throw error;
    }
  }
  
  // 验证响应
  validateResponse(response) {
    // 检查CSP头部
    const cspHeader = response.headers.get('Content-Security-Policy');
    if (!cspHeader) {
      console.warn('响应缺少CSP头部');
    }
    
    // 检查其他安全头部
    const securityHeaders = [
      'X-Content-Type-Options',
      'X-Frame-Options',
      'X-XSS-Protection'
    ];
    
    securityHeaders.forEach(header => {
      if (!response.headers.get(header)) {
        console.warn(`响应缺少安全头部: ${header}`);
      }
    });
    
    return true;
  }
  
  // 获取CSRF Token
  getCSRFToken() {
    const meta = document.querySelector('meta[name="csrf-token"]');
    return meta ? meta.content : '';
  }
  
  // 安全的WebSocket连接
  secureWebSocket(url) {
    // 验证WebSocket URL
    if (!url.startsWith('wss://')) {
      throw new Error('WebSocket必须使用WSS协议');
    }
    
    // 验证域名
    const wsUrl = new URL(url);
    const isWhitelisted = this.apiWhitelist.some(allowed =>
      wsUrl.hostname === new URL(allowed).hostname
    );
    
    if (!isWhitelisted) {
      throw new Error(`未授权的WebSocket地址: ${url}`);
    }
    
    return new WebSocket(url);
  }
  
  // 数据加密传输
  async encryptAndSend(url, data) {
    // 使用Web Crypto API加密数据
    const encryptedData = await this.encryptData(data);
    
    return this.secureRequest(url, {
      method: 'POST',
      body: JSON.stringify({
        encrypted: true,
        data: encryptedData
      })
    });
  }
  
  // 简单的数据加密（实际应用中应使用更强的加密）
  async encryptData(data) {
    const encoder = new TextEncoder();
    const dataBuffer = encoder.encode(JSON.stringify(data));
    
    // 生成密钥（实际应用中应从服务器获取）
    const key = await crypto.subtle.generateKey(
      { name: 'AES-GCM', length: 256 },
      true,
      ['encrypt', 'decrypt']
    );
    
    // 生成IV
    const iv = crypto.getRandomValues(new Uint8Array(12));
    
    // 加密数据
    const encrypted = await crypto.subtle.encrypt(
      { name: 'AES-GCM', iv },
      key,
      dataBuffer
    );
    
    return {
      iv: Array.from(iv),
      data: Array.from(new Uint8Array(encrypted))
    };
  }
}
```

// src/utils/auth/security.js - 登录安全

class LoginSecurity {
  constructor() {
    this.maxAttempts = 5; // 最大尝试次数
    this.lockoutTime = 15 * 60 * 1000; // 锁定15分钟
    this.attempts = this.loadAttempts();
  }
  
  // 加载尝试记录
  loadAttempts() {
    try {
      const stored = localStorage.getItem('login_attempts');
      return stored ? JSON.parse(stored) : {};
    } catch {
      return {};
    }
  }
  
  // 保存尝试记录
  saveAttempts() {
    localStorage.setItem('login_attempts', JSON.stringify(this.attempts));
  }
  
  // 检查是否被锁定
  isLockedOut(identifier) {
    const attempts = this.attempts[identifier];
    if (!attempts) return false;
    
    const { count, lastAttempt } = attempts;
    const now = Date.now();
    
    // 检查是否超过最大尝试次数
    if (count >= this.maxAttempts) {
      // 检查锁定时间是否已过
      if (now - lastAttempt < this.lockoutTime) {
        const remainingTime = Math.ceil((this.lockoutTime - (now - lastAttempt)) / 1000);
        throw new Error(`账户已锁定，请${remainingTime}秒后再试`);
      } else {
        // 锁定时间已过，重置计数
        delete this.attempts[identifier];
        this.saveAttempts();
        return false;
      }
    }
    
    return false;
  }
  
  // 记录登录尝试
  recordAttempt(identifier, success) {
    if (!this.attempts[identifier]) {
      this.attempts[identifier] = { count: 0, lastAttempt: Date.now() };
    }
    
    if (success) {
      // 登录成功，清除记录
      delete this.attempts[identifier];
    } else {
      // 登录失败，增加计数
      this.attempts[identifier].count++;
      this.attempts[identifier].lastAttempt = Date.now();
    }
    
    this.saveAttempts();
  }
  
  // 验证登录
  async validateLogin(identifier, password) {
    // 检查是否被锁定
    this.isLockedOut(identifier);
    
    try {
      // 调用登录API
      const response = await fetch('/api/login', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'X-Login-Attempt': identifier // 用于记录尝试
        },
        body: JSON.stringify({ identifier, password })
      });
      
      const result = await response.json();
      
      if (response.ok && result.success) {
        // 登录成功
        this.recordAttempt(identifier, true);
        
        // 安全存储登录信息
        secureStorage.setItem('auth_token', result.token, {
          encrypted: true,
          expireTime: 24 * 60 * 60 * 1000 // 24小时
        });
        
        return { success: true, user: result.user };
      } else {
        // 登录失败
        this.recordAttempt(identifier, false);
        throw new Error(result.message || '登录失败');
      }
    } catch (error) {
      console.error('登录验证失败:', error);
      throw error;
    }
  }
  
  // 登出
  async logout() {
    try {
      // 调用登出API
      await fetch('/api/logout', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${secureStorage.getItem('auth_token')}`
        }
      });
    } catch (error) {
      console.error('登出失败:', error);
    } finally {
      // 清除本地存储
      secureStorage.removeItem('auth_token');
      sessionStorage.removeItem('user_info');
    }
  }
  
  // 检查登录状态
  isAuthenticated() {
    const token = secureStorage.getItem('auth_token');
    return !!token;
  }
  
  // 获取认证token
  getAuthToken() {
    return secureStorage.getItem('auth_token');
  }
  
  // 刷新token
  async refreshToken() {
    try {
      const response = await fetch('/api/token/refresh', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${this.getAuthToken()}`
        }
      });
      
      if (response.ok) {
        const result = await response.json();
        secureStorage.setItem('auth_token', result.token, {
          encrypted: true,
          expireTime: 24 * 60 * 60 * 1000
        });
        return result.token;
      } else {
        // 刷新失败，需要重新登录
        this.logout();
        throw new Error('Token刷新失败，请重新登录');
      }
    } catch (error) {
      console.error('Token刷新失败:', error);
      throw error;
    }
  }
}

class APISecurity {
  constructor() {
    this.rateLimiter = new Map();
    this.maxRequests = 100; // 每分钟最大请求数
    this.windowMs = 60 * 1000; // 时间窗口（1分钟）
  }
  
  // 速率限制
  checkRateLimit(identifier) {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    
    if (!this.rateLimiter.has(identifier)) {
      this.rateLimiter.set(identifier, []);
    }
    
    const requests = this.rateLimiter.get(identifier);
    
    // 清理过期请求
    const validRequests = requests.filter(time => time > windowStart);
    
    if (validRequests.length >= this.maxRequests) {
      throw new Error('请求过于频繁，请稍后再试');
    }
    
    // 添加当前请求
    validRequests.push(now);
    this.rateLimiter.set(identifier, validRequests);
    
    return true;
  }
  
  // API签名
  generateSignature(method, url, data, timestamp) {
    const message = `${method.toUpperCase()}${url}${JSON.stringify(data || {})}${timestamp}`;
    const secret = secureStorage.getItem('api_secret');
    
    // 使用HMAC-SHA256生成签名
    return this.hmacSHA256(message, secret);
  }
  
  // 简单的HMAC实现（实际应用中应使用成熟的加密库）
  hmacSHA256(message, secret) {
    // 这里使用简化的实现
    // 实际应用中应使用crypto-js或Web Crypto API
    return btoa(message + secret);
  }
  
  // 安全的API请求
  async secureAPIRequest(method, url, data = null) {
    const timestamp = Date.now();
    const signature = this.generateSignature(method, url, data, timestamp);
    
    // 获取用户标识（IP或用户ID）
    const identifier = this.getIdentifier();
    
    // 检查速率限制
    this.checkRateLimit(identifier);
    
    const options = {
      method,
      headers: {
        'Content-Type': 'application/json',
        'X-Request-Timestamp': timestamp.toString(),
        'X-Request-Signature': signature,
        'X-Client-Version': '1.0.0',
        'Authorization': `Bearer ${loginSecurity.getAuthToken()}`
      }
    };
    
    if (data) {
      options.body = JSON.stringify(data);
    }
    
    try {
      const response = await fetch(url, options);
      
      // 验证响应签名
      if (!this.validateResponseSignature(response)) {
        throw new Error('响应签名验证失败');
      }
      
      return response;
    } catch (error) {
      console.error('API请求失败:', error);
      throw error;
    }
  }
  
  // 获取用户标识
  getIdentifier() {
    // 优先使用用户ID，否则使用IP地址
    const userId = sessionStorage.getItem('user_id');
    if (userId) return userId;
    
    // 实际应用中应该从后端获取真实IP
    return 'anonymous_' + this.getClientFingerprint();
  }
  
  // 获取客户端指纹
  getClientFingerprint() {
    // 简单的客户端指纹生成
    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');
    ctx.textBaseline = 'top';
    ctx.font = '14px Arial';
    ctx.fillText('fingerprint', 2, 2);
    
    const fingerprint = canvas.toDataURL();
    return btoa(fingerprint).slice(0, 16);
  }
  
  // 验证响应签名
  validateResponseSignature(response) {
    const responseSignature = response.headers.get('X-Response-Signature');
    const timestamp = response.headers.get('X-Response-Timestamp');
    
    if (!responseSignature || !timestamp) {
      console.warn('响应缺少签名信息');
      return false;
    }
    
    // 检查时间戳（防止重放攻击）
    const now = Date.now();
    const responseTime = parseInt(timestamp);
    
    if (Math.abs(now - responseTime) > 5 * 60 * 1000) { // 5分钟容差
      console.warn('响应时间戳异常');
      return false;
    }
    
    return true;
  }
  
  // 参数验证
  validateParams(params, schema) {
    for (const [key, rules] of Object.entries(schema)) {
      const value = params[key];
      
      // 必填检查
      if (rules.required && (value === undefined || value === null || value === '')) {
        throw new Error(`参数${key}不能为空`);
      }
      
      // 类型检查
      if (value !== undefined && rules.type) {
        const actualType = typeof value;
        if (actualType !== rules.type) {
          throw new Error(`参数${key}类型错误，期望${rules.type}，实际${actualType}`);
        }
      }
      
      // 长度检查
      if (value && rules.minLength && value.length < rules.minLength) {
        throw new Error(`参数${key}长度不能少于${rules.minLength}`);
      }
      
      if (value && rules.maxLength && value.length > rules.maxLength) {
        throw new Error(`参数${key}长度不能超过${rules.maxLength}`);
      }
      
      // 范围检查
      if (value && rules.min && value < rules.min) {
        throw new Error(`参数${key}不能小于${rules.min}`);
      }
      
      if (value && rules.max && value > rules.max) {
        throw new Error(`参数${key}不能大于${rules.max}`);
      }
      
      // 正则表达式验证
      if (value && rules.pattern && !rules.pattern.test(value)) {
        throw new Error(`参数${key}格式不正确`);
      }
    }
    
    return true;
  }
}

```javascript
          description: check.description,
          severity: check.severity
        })),
        warnings: results.warnings.map(check => ({
          id: check.id,
          category: check.category,
          description: check.description,
          error: check.error
        }))
      },
      recommendations: this.generateRecommendations(results)
    };
    
    return report;
  }
  
  // 生成改进建议
  generateRecommendations(results) {
    const recommendations = [];
    
    if (results.failed.length > 0) {
      const highSeverity = results.failed.filter(check => check.severity === 'high');
      if (highSeverity.length > 0) {
        recommendations.push({
          priority: 'high',
          issue: '存在高风险安全问题',
          action: '立即修复所有高风险安全问题，特别是XSS和CSRF防护'
        });
      }
      
      recommendations.push({
        priority: 'medium',
        issue: '安全检查未通过',
        action: '根据失败项目的严重程度，制定修复计划并优先处理'
      });
    }
    
    if (results.warnings.length > 0) {
      recommendations.push({
        priority: 'low',
        issue: '检查过程中出现警告',
        action: '检查警告原因，确保不会影响安全检测的准确性'
      });
    }
    
    if (results.passed.length === this.checks.length) {
      recommendations.push({
        priority: 'info',
        issue: '所有安全检查通过',
        action: '继续保持良好的安全实践，定期更新安全检查清单'
      });
    }
    
    return recommendations;
  }
```

```javascript
// src/utils/security/vulnerability-scanner.js

class VulnerabilityScanner {
  constructor() {
    this.scanners = [
      this.scanForXSS,
      this.scanForCSRF,
      this.scanForInjection,
      this.scanForInsecureStorage,
      this.scanForWeakEncryption
    ];
  }
  
  // 扫描XSS漏洞
  scanForXSS() {
    const vulnerabilities = [];
    
    // 检查innerHTML使用
    const scripts = document.querySelectorAll('script');
    scripts.forEach(script => {
      if (script.textContent.includes('innerHTML')) {
        vulnerabilities.push({
          type: 'XSS',
          severity: 'high',
          location: script.src || 'inline script',
          description: '发现innerHTML使用，可能存在XSS风险'
        });
      }
    });
    
    return vulnerabilities;
  }
  
  // 扫描CSRF漏洞
  scanForCSRF() {
    const vulnerabilities = [];
    
    // 检查表单是否有CSRF token
    const forms = document.querySelectorAll('form');
    forms.forEach(form => {
      const hasCSRFToken = form.querySelector('input[name="csrf_token"], input[name="_token"]');
      if (!hasCSRFToken) {
        vulnerabilities.push({
          type: 'CSRF',
          severity: 'high',
          location: form.action,
          description: '表单缺少CSRF token保护'
        });
      }
    });
    
    return vulnerabilities;
  }
  
  // 扫描注入漏洞
  scanForInjection() {
    const vulnerabilities = [];
    
    // 检查SQL注入风险
    const inputs = document.querySelectorAll('input, textarea');
    inputs.forEach(input => {
      if (!input.hasAttribute('pattern') && input.type !== 'hidden') {
        vulnerabilities.push({
          type: 'Injection',
          severity: 'medium',
          location: input.name || input.id,
          description: '输入缺少验证模式'
        });
      }
    });
    
    return vulnerabilities;
  }
  
  // 扫描不安全存储
  scanForInsecureStorage() {
    const vulnerabilities = [];
    
    // 检查localStorage中的敏感数据
    for (let i = 0; i < localStorage.length; i++) {
      const key = localStorage.key(i);
      const value = localStorage.getItem(key);
      
      if (this.isSensitiveData(key, value)) {
        vulnerabilities.push({
          type: 'InsecureStorage',
          severity: 'high',
          location: `localStorage.${key}`,
          description: '发现敏感数据存储在localStorage中'
        });
      }
    }
    
    return vulnerabilities;
  }
  
  // 扫描弱加密
  scanForWeakEncryption() {
    const vulnerabilities = [];
    
    // 检查是否使用弱加密算法
    const scripts = document.querySelectorAll('script');
    scripts.forEach(script => {
      if (script.textContent.includes('MD5') || 
          script.textContent.includes('SHA1') ||
          script.textContent.includes('DES')) {
        vulnerabilities.push({
          type: 'WeakEncryption',
          severity: 'high',
          location: script.src || 'inline script',
          description: '发现使用弱加密算法'
        });
      }
    });
    
    return vulnerabilities;
  }
  
  // 判断是否为敏感数据
  isSensitiveData(key, value) {
    const sensitivePatterns = [
      /password/i,
      /credit.*card/i,
      /ssn/i,
      /token/i,
      /secret/i,
      /private.*key/i
    ];
    
    return sensitivePatterns.some(pattern => 
      pattern.test(key) || pattern.test(value)
    );
  }
  
  // 运行所有扫描
  async runAllScans() {
    const allVulnerabilities = [];
    
    for (const scanner of this.scanners) {
      try {
        const vulnerabilities = await scanner.call(this);
        allVulnerabilities.push(...vulnerabilities);
      } catch (error) {
        console.error(`扫描器 ${scanner.name} 失败:`, error);
      }
    }
    
    return {
      vulnerabilities: allVulnerabilities,
      summary: {
        total: allVulnerabilities.length,
        bySeverity: this.groupBySeverity(allVulnerabilities),
        byType: this.groupByType(allVulnerabilities)
      }
    };
  }
  
  // 按严重程度分组
  groupBySeverity(vulnerabilities) {
    return vulnerabilities.reduce((groups, vuln) => {
      const severity = vuln.severity;
      if (!groups[severity]) {
        groups[severity] = 0;
      }
      groups[severity]++;
      return groups;
    }, {});
  }
  
  // 按类型分组
  groupByType(vulnerabilities) {
    return vulnerabilities.reduce((groups, vuln) => {
      const type = vuln.type;
      if (!groups[type]) {
        groups[type] = 0;
      }
      groups[type]++;
      return groups;
    }, {});
  }
}
```

### Q1: 解释一下XSS攻击的原理和防护措施

**回答要点**:
- XSS定义：跨站脚本攻击，攻击者注入恶意脚本
- 三种类型：存储型、反射型、DOM型
- 防护措施：输入验证、输出编码、CSP、HttpOnly Cookie

### Q2: 如何防止CSRF攻击？

**回答要点**:
- CSRF定义：跨站请求伪造，利用用户身份执行非本意操作
- 防护方法：CSRF Token、SameSite Cookie、双重提交Cookie、验证Referer

### Q3: 前端如何安全地存储敏感数据？

**回答要点**:
- 避免在localStorage/sessionStorage存储敏感数据
- 使用安全的存储机制（加密、过期时间）
- 敏感数据应存储在服务器端
- 使用HTTPS传输

### Q4: 什么是内容安全策略(CSP)？

**回答要点**:
- CSP定义：HTTP响应头，用于控制页面可以加载哪些资源

- 配置方式：meta标签或HTTP头部
- 常用指令：default-src、script-src、style-src、img-src

```javascript
  // 简单的签名示例

function validateResponseSignature(response) {
  // 获取响应头中的签名
  const responseSignature = response.headers['x-response-signature']
  const timestamp = response.headers['x-response-timestamp']
  
  if (!responseSignature || !timestamp) {
    return false
  }
  
  // 检查时间戳（防止重放攻击）
  const now = Date.now()
  if (Math.abs(now - parseInt(timestamp)) > 300000) { // 5分钟容错
    return false
  }
  
  // 验证签名（简化版本）
  const expectedSignature = btoa(`${response.status}:${timestamp}`)
  return responseSignature === expectedSignature
```

```javascript
export function validateImageUrl(url, options = {}) {
  const { allowExternal = false } = options
  
  try {
    const urlObj = new URL(url)
    
    // 检查协议
    if (!['http:', 'https:', 'data:'].includes(urlObj.protocol)) {
      return {
        isValid: false,
        error: '不支持的协议类型'
      }
    }
    
    // 检查域名白名单
    if (!allowExternal) {
      const allowedDomains = [
        'localhost',
        'liuma-platform.com',
        'cdn.liuma-platform.com'
      ]
      
      if (!allowedDomains.includes(urlObj.hostname)) {
        return {
          isValid: false,
          error: '不允许的外部图片域名'
        }
      }
    }
    
    // 检查文件扩展名
    const allowedExtensions = ['jpg', 'jpeg', 'png', 'gif', 'svg', 'webp']
    const extension = urlObj.pathname.split('.').pop().toLowerCase()
    
    if (!allowedExtensions.includes(extension)) {
      return {
        isValid: false,
        error: '不支持的图片格式'
      }
    }
    
    return {
      isValid: true,
      url: url
    }
  } catch (error) {
    return {
      isValid: false,
      error: '无效的URL格式'
    }
  }
}
```
